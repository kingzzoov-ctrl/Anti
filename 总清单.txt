十、 匹配引擎的数学推演与 SQL 落地 (Resonance Engine)
在陌生人发现 (Discovery) 模块中，匹配引擎不仅依赖语义，还必须结合 7 维心理特征 V_1 到 V_7 进行计算。底层的共鸣匹配度 S 与疲劳衰减公式如下：
在应用层，这套逻辑需要被转化为精准的 SQL 检索，筛选出符合条件的候选人：
-- 核心匹配与疲劳衰减查询 (示例: 查找与当前用户 user_A 最契合的 Top 20)
WITH target_user AS (
    SELECT v_embedding, v_feature FROM insight_reports WHERE user_id = 'user_A_uuid' LIMIT 1
)
SELECT 
    r.user_id,
    -- 计算 1536 维余弦相似度
    1 - (r.v_embedding <=> t.v_embedding) AS semantic_similarity,
    -- 结合曝光日志计算衰减惩罚 (假设 α=0.05)
    (1 - (r.v_embedding <=> t.v_embedding)) * GREATEST(0, 1 - (COALESCE(e.daily_exposure_count, 0) * 0.05)) AS final_score
FROM 
    insight_reports r
CROSS JOIN 
    target_user t
LEFT JOIN 
    exposure_logs e ON r.user_id = e.user_id AND e.date = CURRENT_DATE
WHERE 
    r.user_id != 'user_A_uuid'
    AND r.is_public = TRUE
    AND r.consistency_score >= 0.6 -- 信度惩罚：剔除逻辑自洽度极低的画像
ORDER BY 
    final_score DESC
LIMIT 20;

十一、 Redis 商业限流引擎 (Lua Token Bucket)
为了严格遵循并发计费禁令并阻断算力白嫖，必须使用 Redis Lua 脚本在网关层执行原子性的 Token 扣减。
-- Redis Lua Script: 动态 Token 扣减与拦截
-- KEYS[1]: user_token_key (e.g., "user:tokens:uuid")
-- ARGV[1]: 扣减额度 (e.g., 10)
-- ARGV[2]: 最小保留阈值 (e.g., 0)

local token_key = KEYS[1]
local cost = tonumber(ARGV[1])
local min_balance = tonumber(ARGV[2])

local current_balance = tonumber(redis.call('GET', token_key) or "0")

if current_balance - cost >= min_balance then
    redis.call('DECRBY', token_key, cost)
    return {1, current_balance - cost} -- 扣费成功，返回剩余余额
else
    return {0, current_balance} -- 余额不足，直接拒绝
end

                full_response_content += chunk
                
                # 为了极致的用户体验，我们可以将原始 chunk 直接推给前端
                # 前端去解析增量的 JSON 字符串，或者后端在这里做缓冲解析
                # 这里提供一种简单的 SSE 格式推送
                yield f"data: {json.dumps({'type': 'chunk', 'content': chunk})}\n\n"
            
            # 流式传输结束后，解析最终的完整 JSON 结构
            try:
                parsed_response = json.loads(full_response_content)
                reply_text = parsed_response.get("reply_to_user", "")
                new_state = parsed_response.get("internal_state", {})
                
                # 将 AI 的回复追加进历史记录 (只存文本，不存内部思考，节省 Token)
                context["messages"].append({"role": "assistant", "content": reply_text})
                # 更新状态机
                context["internal_state"] = new_state
                
                # 发送状态机跃迁事件给前端，触发 UI 变色或动效
                yield f"data: {json.dumps({'type': 'state_mutation', 'state': new_state})}\n\n"
                
            except json.JSONDecodeError:
                # 容错机制：如果大模型没有遵守 JSON 格式，尝试强行提取文本
                context["messages"].append({"role": "assistant", "content": full_response_content})
            
            # 6. 持久化回写：将更新后的 25 轮完整上下文塞回 Redis，TTL 设为 24 小时
            await redis.setex(session_key, 86400, json.dumps(context))

            # 7. 如果本轮对话后触发了收敛，发送结束信号
            if context["internal_state"].get("is_ready_for_report"):
                yield f"data: {json.dumps({'type': 'system_signal', 'action': 'FINISH_AND_GENERATE'})}\n\n"

        except Exception as e:
            yield f"data: {json.dumps({'type': 'error', 'message': '基座神经元连接波动，请重试。'})}\n\n"

    return StreamingResponse(event_generator(), media_type="text/event-stream")

代码落地要点解析：
 * 防爆栈设计： INTERVIEW_SYSTEM_PROMPT 只在后端启动时读入内存一次，后续所有并发请求共享这一个字符串，不会因为高并发把服务器内存撑爆。
 * 纯粹的无状态： 注意到 context 是完全从 Redis 中反序列化出来的，处理完后又塞回去了。这意味着你的 FastAPI 节点可以随时通过 K8s 扩容到 100 个实例，请求打到任何一台机器上都能无缝衔接。
 * SSE 状态分离： yield 的数据被分为了 'type': 'chunk'（用于打字机文字展示）和 'type': 'state_mutation'（用于前端捕捉大模型的内部矛盾追踪状态并改变页面 UI 压迫感）。
这份网关代码已经把最难处理的“流式状态同步”和“纯 Prompt 接管”解决了。

十四、 自动化洞察报告流水线 (Report Generation Task)
当对话基座抛出 FINISH_AND_GENERATE 信号后，FastAPI 必须立刻切断长连接，转入后台异步任务（建议使用 BackgroundTasks 或 Celery）。该任务将负责加载 report.skill.md，提炼 7 维特征，并生成最终向量库资产。
# services/report_pipeline.py
import json
import os
from sqlalchemy.ext.asyncio import AsyncSession
from core.llm_client import call_llm_structured # 支持强制 JSON 结构输出的 LLM 客户端
from core.vector_api import get_embeddings # 调用如 text-embedding-3-small
from models.database import InsightReport, UserProfile

# 加载报告生成的 Prompt 资产
REPORT_SKILL_PATH = os.path.join(os.path.dirname(__file__), "../skills/report.skill.md")
with open(REPORT_SKILL_PATH, "r", encoding="utf-8") as f:
    REPORT_SYSTEM_PROMPT = f.read()

async def generate_insight_report_task(user_id: str, redis_client, db_session: AsyncSession):
    """
    后台任务：提炼最终洞察报告并落库
    """
    session_key = f"ariadne:session:{user_id}"
    session_data = await redis_client.get(session_key)
    if not session_data:
        return
        
    context = json.loads(session_data)
    chat_history = context.get("messages", [])
    
    # 1. 组装提炼请求：将 25 轮对话记录作为 User 扔给大模型
    # 强制 LLM 按照 report.skill.md 的要求输出结构化 JSON，必须包含 Markdown 正文和 7 维向量分值
    prompt_payload = [
        {"role": "system", "content": REPORT_SYSTEM_PROMPT},
        {"role": "user", "content": f"请基于以下完整问询记录，生成最终动力学洞察报告：\n{json.dumps(chat_history, ensure_ascii=False)}"}
    ]
    
    # 2. 执行高优推理 (必须使用 Claude Opus 或 GPT-4 级别)
    report_data = await call_llm_structured(prompt_payload)
    
    # report_data 预期结构: 
    # {
    #   "markdown_content": "# 您的核心动力学洞察...",
    #   "v_feature_7d": [0.8, -0.4, 0.9, 0.1, 0.5, -0.2, 0.7],
    #   "consistency_score": 0.85
    # }
    
    # 3. 语义降维：为全文生成 1536 维向量，供 pgvector 快速匹配使用
    semantic_embedding = await get_embeddings(report_data["markdown_content"])
    
    # 4. 数据落库
    new_report = InsightReport(
        user_id=user_id,
        raw_content=report_data["markdown_content"],
        v_feature=report_data["v_feature_7d"],
        v_embedding=semantic_embedding,
        consistency_score=report_data["consistency_score"],
        is_public=True # 默认进入匹配池
    )
    db_session.add(new_report)
    await db_session.commit()
    
    # 5. 清理 Redis 占用，并触发分发引擎
    await redis_client.delete(session_key)
    await dispatch_notifications(user_id, new_report.id, db_session)

# services/notification_dispatch.py
import json

async def dispatch_notifications(user_id: str, report_id: str, db_session):

    # channels 格式例: {"email": ["a@test.com", "b@test.com"], "wechat": ["wxid_123"], "telegram": ["@user"]}
    channels = user.notification_channels 
    
    if user.tier in ["Free", "Ad-Reward"]:

router = APIRouter(prefix="/api/v1/ariadne", tags=["Resonance Matching"])

MATCH_SKILL_PATH = os.path.join(os.path.dirname(__file__), "../../skills/match.skill.md")

@router.post("/match/deep_resonance")
async def execute_deep_match(
    target_user_id: str,
    current_user_id: str = Depends(get_current_user),
    db: AsyncSession = Depends(get_db_session),
    redis = Depends(get_redis_client)
):
    """
    第二阶段：调用 match.skill.md 执行深度的灵魂火药桶推演
    """
    # 1. 高阶操作，单次推演扣除高额 Token (例如 50 Token)
    if not await check_and_deduct_token(redis, current_user_id, cost=50):
        raise HTTPException(status_code=402, detail="Token 余额不足，无法执行深度张力推演。")
        
    # 2. 从数据库拉取双方的底层动力学报告 (Markdown 原文)
    user_report = await get_latest_report(db, current_user_id)
    target_report = await get_latest_report(db, target_user_id)
    
    if not user_report or not target_report:
        raise HTTPException(status_code=404, detail="底层报告缺失，无法进行推演。")
        
    # 3. 构建高维碰撞的上下文
    # 严格按照 match.skill.md 的设定，将两份报告作为输入，让大模型在脑内碰撞
    match_payload = [
        {"role": "system", "content": MATCH_SYSTEM_PROMPT},
        {"role": "user", "content": f"【用户 A 画像】：\n{user_report.raw_content}\n\n【用户 B 画像】：\n{target_report.raw_content}\n\n请执行深度推演，并输出包含核心共鸣点与潜在火药桶的完整避坑说明书。"}
    ]
    
    # 4. 返回流式结果给前端呈现“推演过程”
    async def match_generator():
        async for chunk in call_llm_streaming(match_payload):
            yield f"data: {chunk}\n\n"
        yield "data: [DONE]\n\n"
        
    return StreamingResponse(match_generator(), media_type="text/event-stream")
